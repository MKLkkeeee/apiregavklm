API Documentation

Integrate Nexus Drive into your applications.

GET /api/list?path={path}

List files and directories in a specific path.

Headers required

Authorization: Bearer <token>
POST /api/file

Upload multiple files. Requires multipart/form-data.

Body (FormData)

path: "/images"
files: [File Object]

JavaScript SDK Reference

storage-sdk.js 
class StorageSDK {
    constructor(token) {
        this.token = token;
        this.baseURL = window.location.origin;
    }

    async _fetch(endpoint, options = {}) {
        const headers = { 'Authorization': `Bearer ${this.token}` };
        if (!(options.body instanceof FormData)) {
            headers['Content-Type'] = 'application/json';
            if (options.body) options.body = JSON.stringify(options.body);
        }
        const res = await fetch(`${this.baseURL}${endpoint}`, { ...options, headers });
        if (!res.ok) throw new Error((await res.json()).error);
        return res.json();
    }

    list(path = '/') { return this._fetch(`/api/list?path=${encodeURIComponent(path)}`); }
    createFolder(path) { return this._fetch('/api/folder', { method: 'POST', body: { path } }); }
    delete(paths) { return this._fetch('/api/delete', { method: 'POST', body: { paths } }); }
    move(items, dest) { return this._fetch('/api/move', { method: 'POST', body: { items, destination: dest } }); }
    
    upload(files, path = '/') {
        const formData = new FormData();
        formData.append('path', path);
        Array.from(files).forEach(f => formData.append('files', f));
        return this._fetch('/api/file', { method: 'POST', body: formData });
    }
}
